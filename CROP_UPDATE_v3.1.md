final overlayPaint = Paint()..color = Colors.black.withOpacity(0.5);

### 2. تحديث `_onPanStart`:

```dart
// القديم (v3.0):
void _onPanStart(DragStartDetails details) {
  setState(() {
    _isDragging = true;
    _dragStart = details.localPosition;
    _dragEnd = details.localPosition;
  });
}

// الجديد (v3.1):
void _onPanStart(DragStartDetails details) {
  // حساب موضع اللمس
  final touchX = convertToImageCoords(details.localPosition.dx);
  final touchY = convertToImageCoords(details.localPosition.dy);
  
  // التحقق من اللمس داخل المنطقة الحالية
  final isInsideCrop = touchX >= _x && touchX <= (_x + _width) &&
                       touchY >= _y && touchY <= (_y + _height);
  
  setState(() {
    _isDragging = true;
    if (isInsideCrop) {
      // البدء من المنطقة الحالية
      _dragStart = details.localPosition;
      _dragEnd = details.localPosition;
    } else {
      // بدء منطقة جديدة
      _dragStart = details.localPosition;
      _dragEnd = details.localPosition;
    }
  });
}
```
